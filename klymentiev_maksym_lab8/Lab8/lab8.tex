    \begin{frame}{Вступ та актуальність}
        \begin{itemize}
            \item \textbf{Проблема:} Колоризація — математично невизначена задача "один до багатьох" (багато кольорів відповідають одній яскравості).
            \item \textbf{Застосування:} Відновлення архівних матеріалів, реставрація, творчі пайплайни.
            \item \textbf{Недоліки існуючих рішень:}
            \begin{itemize}
                \item Відсутність гнучкого контролю користувача.
                \item Ефект "сепії" у старих моделях.
                \item Часова неузгодженість (мерехтіння) у відео.
            \end{itemize}
            \item \textbf{Мета:} Створення інтерактивної системи з багатоваріантною генерацією на базі сучасних генеративних моделей.
        \end{itemize}
    \end{frame}

    \begin{frame}{Інтерактивна взаємодія та відео}
        \begin{block}{Типи взаємодії}
            \begin{itemize}
                \item \textbf{Scribble-based:} Кольорові точки та лінії як локальні підказки.
                \item \textbf{Language-based:} Використання CLIP для текстового керування стилем.
                \item \textbf{Exemplar-based:} Перенесення палітри з референсного зображення.
            \end{itemize}
        \end{block}
        
        \begin{block}{Особливості обробки відео}
            \begin{itemize}
                \item Використання \textit{Deep Feature Propagation} для перенесення ознак між кадрами.
                \item Саморегуляризація для усунення мерехтіння.
            \end{itemize}
        \end{block}
    \end{frame}

    \begin{frame}{Висновки}
        \begin{itemize}
            \item Проведено аналіз сучасних методів колоризації: класичних, CNN, GAN та дифузійних моделей.
            \item Виявлено, що гібридні дифузійні моделі є найбільш перспективними для забезпечення керованості та реалізму.
            \item Сформовано архітектуру майбутньої системи, яка об'єднає мультимодальні вхідні дані (текст, маски) для багатоваріантної генерації.
            \item Наступний етап — програмна реалізація алгоритму та проведення експериментального порівняння метриками PSNR, LPIPS та FID.
        \end{itemize}
    \end{frame}
