            \begin{itemize}

                \item Ефект <<сепії>> та низька насиченість.
                \item Складність вибору при високій невизначеності кольору.
            \end{itemize}
            \vspace{0.3cm} % Додаємо простір для балансу
            \item \textbf{Мета:} Створення інтерактивної системи з багатоваріантною генерацією на базі сучасних моделей.
        \end{itemize}

    \begin{frame}{Інтерактивна взаємодія}
        \textbf{Типи взаємодії:}
        \begin{itemize}
            \item \textbf{Scribble-based:} Використання кольорових штрихів як локальних орієнтирів.
            \item \textbf{Language-based:} Текстові описи для глобальної стилізації.
            \item \textbf{Exemplar-based:} Використання референсного зображення для копіювання палітри.
        \end{itemize}
        
        \vspace{2ex}
        
        \textbf{Забезпечення багатоваріантності:}
        \begin{itemize}
            \item Внесення пертурбацій у простір ознак або у вхідні підказки.
            \item Використання стохастичної природи генеративних моделей.
        \end{itemize}
    \end{frame}

    \begin{frame}{Висновки та подальші плани}
        \begin{itemize}
            \item Проведено аналіз архітектур CNN, GAN, Transformers та дифузійних моделей у задачах колоризації.
            \item Виявлено переваги інтерактивного підходу для усунення візуальної неоднозначності.
            \item Сформовано структуру системи, що підтримує багатоваріантну генерацію на основі користувацьких умов.
            \item \textbf{Наступні кроки:} програмна реалізація обраного алгоритму та оцінка результатів за метриками PSNR, SSIM та FID.
        \end{itemize}
    \end{frame}
